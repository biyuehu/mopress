ејҖе§Ӣ

жң¬иҠӮе°ҶеёҰдҪ д»Һйӣ¶зј–еҶҷдёҖд»ҪжңҖз®ҖеҚ•зҡ„жһ„е»әи„ҡжң¬пјҢжҠҠжәҗзӣ®еҪ•дёӯзҡ„ Markdown ж–Үд»¶жёІжҹ“дёә HTMLгҖӮ

еҮҶеӨҮе·ҘдҪң

иҜ·е…ҲеҸӮиҖғ Book жЁЎејҸзҡ„ еҝ«йҖҹејҖе§Ӣ дёӯзҡ„е®үиЈ… MoPress жӯҘйӘӨгҖӮ

Site жЁЎејҸдёӢпјҢйЎ№зӣ®зҡ„е…ҘеҸЈжҳҜдёҖд»Ҫ MoonBit жәҗж–Үд»¶пјҲж–Үд»¶еҗҚеҸҜд»ҘиҮӘе®ҡд№үпјҢеҸӘиҰҒеңЁиҝҗиЎҢж—¶жӯЈзЎ®жҢҮе®ҡеҚіеҸҜпјүгҖӮйңҖиҰҒеңЁйЎ№зӣ®дёӯеј•е…Ҙж ёеҝғеә“дҪңдёәдҫқиө–гҖӮ

еҲӣе»әйЎ№зӣ®

иҫ“е…Ҙд»ҘдёӢе‘Ҫд»Өпјҡ

mopress new <dir>
cd <dir>

з”ҹжҲҗд»ҘдёӢ <dir> зӣ®еҪ•з»“жһ„пјҡ

в”ңв”Җв”Җ site.mbtx
в”ңв”Җв”Җ index.md
в”ңв”Җв”Җ about.md
в”ңв”Җв”Җ contact.md
в”ңв”Җв”Җ templates
в”Ӯ   в””в”Җв”Җ default.html
в””в”Җв”Җ styles
    в””в”Җв”Җ default.css

е…¶дёӯ site.mbtx дёә

import {
  "himeno/mopress/core",
  "moonbitlang/async",
}

///|
async fn main {
  @core.mopress([
    Glob(
      "**/*.md",
      Text(raw => {
        raw
        |> @core.set_extension(".html")
        |> @core.render_markdown_and_frontmatter
        |> @core.load_and_apply_template("templates/default.html")
        |> @core.use_js(["console.log('hi, moonbit! hi, mopress!')"])
        |> @core.unify
      }),
    ),
    Glob("styles/**/*", Copy),
  ])
}

иҝҷд»Ҫд»Јз ҒеЈ°жҳҺдәҶдёҖжқЎи§„еҲҷпјҡеҢ№й…Қжәҗзӣ®еҪ•дёӢжүҖжңүд»Ҙ .md з»“е°ҫзҡ„ж–Үд»¶пјҢе°Ҷе…¶еҶ…е®№дҪңдёәж–Үжң¬иҜ»е…ҘпјҢдҫқж¬Ўз»ҸиҝҮвҖңжёІжҹ“ Markdown дёҺе…ғж•°жҚ®вҖқе’Ң дҝ®ж”№зӣ®ж Үжү©еұ•еҗҚдёә.htmlвҖқзӯүеӨҡдёӘ stepпјҢжңҖеҗҺи°ғз”Ё unify иҪ¬жҚўдёәжңҖз»ҲеҸҜеҶҷе…ҘзЈҒзӣҳзҡ„ж јејҸгҖӮ

иҝҗиЎҢжһ„е»ә

moon run --target native site.mbtx build

жһ„е»әе®ҢжҲҗеҗҺпјҢжәҗзӣ®еҪ•дёӯзҡ„ hello.markdown дјҡиў«жёІжҹ“дёәиҫ“еҮәзӣ®еҪ• dest дёӯзҡ„ hello.htmlгҖӮ

иҮӘе®ҡд№үжәҗзӣ®еҪ•дёҺиҫ“еҮәзӣ®еҪ•

й»ҳи®Өжғ…еҶөдёӢпјҢжһ„е»әеҮҪж•°дҪҝз”Ёй»ҳи®Өзҡ„ Options жҸҗдҫӣзҡ„жәҗзӣ®еҪ•дёҺиҫ“еҮәзӣ®еҪ•гҖӮеҰӮжһңйңҖиҰҒиҮӘе®ҡд№үпјҢеҸҜд»ҘжҳҫејҸдј е…Ҙпјҡ

///|
async fn main {
  let options :  @core.Options = { src: "./content", dest: "./public" }
  @core.mopress(
    options~,
    // ...
  )
}

еҲқе§ӢеҢ–дёҺе…ұжңүйЎ№

еҪ“йңҖиҰҒеЈ°жҳҺдёҖзі»еҲ—жүҖжңүи§„еҲҷе…ұз”Ёзҡ„еҖјж—¶еҸҜдҪҝз”Ё @core.mopress_with еҮҪж•°пјҡ

pub async fn[E] mopress_with(
  options? : Options = Options::default(),
  init : (Options) -> E,
  rules : (Options, E) -> Rules,
) -> Unit

еңЁдј е…Ҙзҡ„ init еҮҪж•°дёӯиҮӘе®ҡд№үиҝ”еӣһзҡ„зұ»еһӢ EпјҢе№¶е°Ҷе…¶дҪңдёә rules еҮҪж•°зҡ„з¬¬дәҢдёӘеҸӮж•°пјҢжүҖжңүзҡ„и§„еҲҷйғҪе°ҶиғҪи®ҝй—®еҲ° E еҖјпјҡ

///|
async fn main {
  @core.mopress_with(
    (options) => {
      let shared = {
        "title": "My Site",
        "footer": "В© 2026 Himeno Sena",
      }
      shared
    },
    (options, shared) => [
      // ...
    ],
  )
}

Tip

е…ідәҺжЁЎжқҝиҜӯжі•жң¬иә«гҖҒд»ҘеҸҠжЁЎжқҝеҸҳйҮҸжҳҜеҰӮдҪ•еҜ№еә”иө·жқҘзҡ„пјҢиҜ·еҸӮи§Ғ з»„еҗҲ Steps дёҺ жЁЎжқҝеј•ж“Һ зӣёе…іиҜҙжҳҺгҖӮ
